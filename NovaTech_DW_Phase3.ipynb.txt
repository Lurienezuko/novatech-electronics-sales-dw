        "def explode_addons(row):\n",
        "    rows = []\n",
        "\n",
        "    # 1. Always include the main product row\n",
        "    main = row.to_dict()\n",
        "    main['is_addon'] = False\n",
        "    main['addon_type'] = None\n",
        "    rows.append(main)\n",
        "\n",
        "    # 2. Handle add-ons if they exist\n",
        "    addons_str = row['Add-ons Purchased']\n",
        "    addon_total = row['Add-on Total'] if pd.notna(row['Add-on Total']) else 0.0\n",
        "\n",
        "    if pd.notna(addons_str) and isinstance(addons_str, str) and addons_str.strip():\n",
        "        addons_list = [a.strip() for a in addons_str.split(',') if a.strip()]\n",
        "        if addons_list:\n",
        "            # Distribute the total addon amount equally across add-ons\n",
        "            split_value = addon_total / len(addons_list)\n",
        "\n",
        "            for addon in addons_list:\n",
        "                addon_row = row.to_dict()\n",
        "                addon_row['Product Type'] = addon              # override with addon name\n",
        "                addon_row['SKU'] = None                        # no SKU for add-ons\n",
        "                addon_row['Unit Price'] = 0.0\n",
        "                addon_row['Total Price'] = split_value\n",
        "                addon_row['Add-on Total'] = split_value        # per addon\n",
        "                addon_row['is_addon'] = True\n",
        "                addon_row['addon_type'] = addon\n",
        "                rows.append(addon_row)\n",
        "\n",
        "    return rows\n",
        "\n",
        "# Apply the explosion to every row\n",
        "exploded_rows = []\n",
        "for idx, row in df.iterrows():\n",
        "    exploded_rows.extend(explode_addons(row))\n",
        "\n",
        "# Create the fact table\n",
        "fact_sales = pd.DataFrame(exploded_rows)\n",
        "\n",

        "# Build dim_customer with SCD Type 2 logic using DuckDB for window functions\n",
        "dim_customer = duckdb.sql(\"\"\"\n",
        "    WITH customer_orders AS (\n",
        "        SELECT DISTINCT\n",
        "            \"Customer ID\" AS customer_id,\n",
        "            age_group,\n",
        "            Gender AS gender,\n",
        "            \"Loyalty Member\" AS loyalty_status,\n",
        "            \"Purchase Date\" AS transaction_date\n",
        "        FROM fact_sales\n",
        "        ORDER BY \"Customer ID\", \"Purchase Date\"\n",
        "    ),\n",
        "    with_versions AS (\n",
        "        SELECT\n",
        "            customer_id,\n",
        "            age_group,\n",
        "            gender,\n",
        "            loyalty_status,\n",
        "            transaction_date AS valid_from,\n",
        "            LEAD(transaction_date) OVER (PARTITION BY customer_id ORDER BY transaction_date) AS valid_to,\n",
        "            ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY transaction_date) AS version\n",
        "        FROM customer_orders\n",
        "        -- Only keep rows where loyalty status actually changes (or first record)\n",
        "        QUALIFY\n",
        "            loyalty_status != LAG(loyalty_status) OVER (PARTITION BY customer_id ORDER BY transaction_date)\n",
        "            OR LAG(loyalty_status) OVER (PARTITION BY customer_id ORDER BY transaction_date) IS NULL\n",
        "    )\n",
        "    SELECT\n",
        "        customer_id,\n",
        "        age_group,\n",
        "        gender,\n",
        "        loyalty_status,\n",
        "        valid_from,\n",
        "        COALESCE(valid_to, '9999-12-31') AS valid_to,\n",
        "        version AS customer_surrogate_key,\n",
        "        CASE WHEN valid_to = '9999-12-31' THEN True ELSE False END AS is_current\n",
        "    FROM with_versions\n",
        "    ORDER BY customer_id, version\n",
        "\"\"\").df()\n",
        "\n",
        "# Validation\n",
